  function closePop(){

  function isEmail(email){
    var regex = /^\w+([\.-]?\w+)*@\w+([\.-]?\w+)*(\.\w{2,3})+$/;
    return regex.test(email);
  }
  function subscribe(){

    var email = $.trim($('#emailNews').val());
    var err = false;

    if((!isEmail(email)) || (email=="")){
      $("#errEmail").css("visibility","visible"); 
      err = true;
    }
    if(err == false){

      $.ajax

        {
          $('#thankyouSub').show();
          $('#formCont').hide();
          $('#existMsg').hide();

            $('#formCont').hide();

          }
        },
        error: function(errorThrown) {
          // console.log('hi');
        }
      });
